data = [41, 12, 6, 6, 13, 22, 63, 130, 176, 184, 154, 182, 168, 234, 178, 200, 192, 278, 228, 227, 254, 183, 161, 63]

cycle = 24
hours = [i for i in range(cycle)]

angles = [(n / float(cycle) * 2 * pi) for n in range(cycle)]
angles += angles[:1] # close loop: repeat last value


fig, ax = plt.subplots(figsize=(10, 10), subplot_kw=dict(polar=True))

plt.xticks(angles[:-1], hours, color='grey', size=10)
ax.set_theta_zero_location("N")
ax.set_theta_direction(-1)
ax.set_rlabel_position(40)

plt.yticks(color="#4a4a4a", size=12)

values = data
values += data[:1] # close loop: repeat last value
ax.plot(angles, values, linewidth=1, linestyle='solid', color=COLOR)
ax.fill(angles, values, COLOR, alpha=0.1)

ax.annotate(
f'I commented {values[22]} times\n         at {hours[22]}:00',
    (angles[22], values[22]),
    (angles[23], 230),
    size=14,
arrowprops=dict(arrowstyle = "-|>", connectionstyle='arc3,rad=0.2', color=COLOR))

ax.set_frame_on(False)
plt.grid(color='#d4d4d4')
plt.title('Number of comments per hour of the day', color='grey', size=16, y=1.1)
plt.show()
