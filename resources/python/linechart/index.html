More customisations¶

In [6]:
fig, ax = plt.subplots(figsize=FIGSIZE)

plt.plot(x, np.sin(x), linestyle='-.', c='black', label='Line label')
plt.axvline(x=0.0, lw=8, alpha=0.5, label='Vertical line')
plt.axhline(y=0.0, linestyle='--', label='Horizontal line')

plt.title('Awesome Title', fontdict=FONT)
plt.xlabel('x: X label', fontdict=FONT)
plt.ylabel('y: Y label', fontdict=FONT)

ax.set_frame_on(False)
plt.grid(True)
plt.legend(prop={'size': FONT['size']})
plt.show()

In [8]:
y1 = np.sin(x)
y2 = np.sin(2*x)

fig, (ax1, ax2, ax3) = plt.subplots(3, 1, figsize=(17,10), sharex=True)

fig.suptitle("Awesome title for whole figure", fontsize=20)

ax1.fill_between(x, y1, 0)
ax1.set_ylabel('between y1 and 0', fontdict=FONT)
ax1.set_title("Title for first plot", fontdict=FONT)
ax1.set_frame_on(False)
ax1.grid(True)

ax2.fill_between(x, y1, 1, color='red', alpha=0.2)
ax2.set_ylabel('between y1 and 1', fontdict=FONT)
ax2.set_title("Title for second plot", fontdict=FONT)


ax3.fill_between(x, y1, y2)
ax3.set_title("Title for third plot", fontdict=FONT)
ax3.set_ylabel('between y1 and y2', fontdict=FONT)
ax3.set_xlabel('x', fontdict=FONT)

plt.grid(True)
plt.show()

import matplotlib.pyplot as plt

x = range(15)
tick_labels = ['label'+str(i) for i in range(len(x))]

plt.plot(x, np.random.rand(len(x)))

ax.set_frame_on(False)
